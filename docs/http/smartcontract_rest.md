> It contains single function in a single file setupsc.go
```sh
func SetupSmartContracts() {
	for _, sc := range scs {
		if viper.GetBool(fmt.Sprintf("server_chain.smart_contract.%v", sc.GetName())) {
			sc.InitSC()
			smartcontract.ContractMap[sc.GetAddress()] = sc
		}
	}
}
```
